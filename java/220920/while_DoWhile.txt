package while_;

public class DoWhile {

	public static void main(String[] args) {
		
		//A B C D E F ~~~~ X Y Z
		//한 줄에 7개씩 출력하시오
		char ch = 'A';
		int count = 0;
		do {
			System.out.print(ch + " ");
			ch++;
			count++;
			if (count%7 == 0) System.out.println();
		}while(ch <= 'Z');

	}

}
